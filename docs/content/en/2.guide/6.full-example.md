## 1. Config

```ts [nuxt.config.ts]
export default defineNuxtConfig({
    modules: ['nuxt-permission'],
    permission: {
        // Only these stay static; every other page becomes dynamic (registered after login)
        static: ['/', '/login'],
    },
})
```

## 2. Backend shape

The server is the source of truth: it returns the current user's flat `permissions` and the `menus` tree. Here `type: 1` is a menu, `type: 2` is a button.

```ts [server/api/user.get.ts]
export default defineEventHandler((event) => {
    const auth = getHeader(event, 'authorization')
    if (!auth?.startsWith('Bearer ')) {
        throw createError({ statusCode: 401, statusMessage: 'Unauthorized' })
    }

    return {
        permissions: ['menu-add', 'menu-edit', 'menu-view'],
        menus: [
            { id: 1, name: 'Dashboard', path: '/dashboard', type: 1, children: [] },
            {
                id: 2,
                name: 'Menu',
                path: '/menu',
                type: 1,
                meta: { sort: 1 },
                children: [
                    { id: 3, name: 'Add', permission: 'menu-add', type: 2 },
                    { id: 4, name: 'Edit', permission: 'menu-edit', type: 2 },
                    { id: 5, name: 'Delete', permission: 'menu-delete', type: 2 },
                ],
            },
        ],
    }
})
```

```ts [server/api/login.post.ts]
// A real app validates credentials; permissions/menus come later from /api/user via the token
export default defineEventHandler(() => {
    return { token: 'a-real-jwt-here' }
})
```

## 3. Token storage (yours)

The library never touches auth. A tiny composable keeps the token reactive and persisted:

```ts [composables/useToken.ts]
export function useToken() {
    // SPA (ssr: false) — token in localStorage
    const token = useState<string>('token', () => (import.meta.client ? localStorage.getItem('token') ?? '' : ''))

    function setToken(value: string) {
        token.value = value
        if (!import.meta.client) return
        value ? localStorage.setItem('token', value) : localStorage.removeItem('token')
    }

    // SSR (ssr: true) — token in a cookie so the server can read it on first paint
    // const token = useCookie<string>('token')
    //
    // function setToken(value: string) {
    //     token.value = value
    // }

    return { token, setToken }
}
```

## 4. Data source

`app/permission.ts` reads the token, fetches from the backend, writes state, and returns the menu tree. The library runs it on first paint/refresh; you run it via `load()` after login.

```ts [app/permission.ts]
export default definePermissionSource(async ({ setPermissionList, setMenuList }) => {
    const { token } = useToken()
    if (!token.value) return [] // not logged in → register nothing

    const { permissions, menus } = await $fetch('/api/user', {
        headers: { authorization: `Bearer ${token.value}` },
    })

    setPermissionList(permissions)

    // Uncomment the next line, sign in as a user with ALL permissions, and paste the line printed in the browser console straight into `PermissionMap` (step 8) for full key autocomplete:
    // console.log('keys: ' + permissions.map(p => `'${p}'`).join(' | '))

    // type 2 = button → mark with _btn so it folds into the parent's meta._permission
    const tree = normalizePermissionMenus(menus, v => ({ ...v, _btn: `${v.type}` === '2' }))
    setMenuList(tree)
    return tree
})
```

## 5. Login & logout

```vue [pages/login.vue]
<script setup lang="ts">
const { setToken } = useToken()
const { load } = usePermissionState()

async function onLogin() {
    const { token } = await $fetch('/api/login', { method: 'POST' })
    setToken(token)
    await load() // run the source, write state, register routes
    await navigateTo('/dashboard')
}
</script>

<template>
    <button @click="onLogin">Login</button>
</template>
```

```ts
// logout, anywhere
const { clear } = usePermissionState()
const { setToken } = useToken()

function onLogout() {
    setToken('')
    clear() // clears state + unregisters the dynamic routes
    navigateTo('/login')
}
```

## 6. Layout: sidebar from `menus`

Render the nav from the reactive `menus` state (set after routes are registered, so links always resolve):

```vue [app.vue]
<script setup lang="ts">
const { menus } = usePermissionState()
const { token } = useToken()
const isLoggedIn = computed(() => !!token.value)
const navMenus = computed(() => menus.value.filter(m => m.path))
</script>

<template>
    <div>
        <nav>
            <NuxtLink to="/">Home</NuxtLink>
            <template v-if="isLoggedIn">
                <NuxtLink v-for="m in navMenus" :key="m.path" :to="m.path">{{ m.name }}</NuxtLink>
            </template>
        </nav>
        <NuxtPage />
    </div>
</template>
```

## 7. Guarding UI with `hasPermission`

Button-level permissions were folded into `meta._permission`; guard with `v-if="hasPermission(...)"`:

```vue [pages/menu.vue]
<script setup lang="ts">
const route = useRoute()
</script>

<template>
    <div>
        <!-- Delete is hidden: 'menu-delete' is not in the granted permissions -->
        <button v-if="hasPermission('menu-add')">
            {{ route.meta?._permission?.['menu-add']?.name }}
        </button>
        <button v-if="hasPermission('menu-edit')">Edit</button>
        <button v-if="hasPermission('menu-delete')">Delete</button>
    </div>
</template>
```

```ts [types.d.ts]
declare module 'nuxt-permission' {

        id: number
        name: string
        permission: string
    }

export {}
```
